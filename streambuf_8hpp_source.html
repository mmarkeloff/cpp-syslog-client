213  m_Mode->lock();
214  auto ready{m_Clnt->isInitialised()};
215  m_Mode->unlock();
216 
217  if (ready) {
218  m_Mode->lock();
219  // https://datatracker.ietf.org/doc/html/rfc5424#section-6.2.1
220  auto pri{"<" + std::to_string((m_Facility << 3) + m_Lvl) + ">"};
221  m_Mode->unlock();
